### Installation

    npm install not-so-human-typist --save

### Usage

    import React, { Component } from 'react';
    import './App.css';
    import Typist from 'not-so-human-typist';

    class App extends Component {
    render() {
        return (
        <div className="App">
                <Typist name='Hi there, I am Donna'/>
        </div>
        );
    }
    }

    export default App;
